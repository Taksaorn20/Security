// แสดงเวอร์ชันของ docker
docker -v

// ทดสอบระบบว่า Docker ในเครื่องติดตั้งเสร็จสมบูรณ์และพร้อมใช้งานหรือไม่
docker run hello-world

// ใช้สำหรับ ตรวจสอบสถานะของ Container ที่กำลังทำงานอยู่ในขณะนี้
docker ps

// การแสดง container ทั้งหมดออกมา
docker ps --all 

// การสั่งให้ Docker ทำงานแบบ "แวบเดียวจบ" เพื่อรันคำสั่งที่เราต้องการข้างใน Container
docker run busybox echo 'hello world in docker'

// การสั่งให้ Docker "ลิสต์รายชื่อไฟล์และโฟลเดอร์" ที่อยู่ภายใน Container ของ BusyBox
docker run busybox ls // ls คือ command ในการใช้ execute
